Demo Code

      <script src="ellipsis.min.js"></script>
      <script type="text/javascript">

        Ellipsis({
          class: '.title',
          lines: 1
        }); //specific conf on titles

        Ellipsis({
          class: '.normal',
          lines: 3,
          portrait: 1
        }) //different maximum lines in portrait/landscape

        Ellipsis({
					break_word: false
				}); //standard usage but we don't want the ellipsis to break words
      </script>
